local ipLabels = {}

local function fetchIPInfo()
    local success, result = pcall(function()
        return game:HttpGet("https://ipwho.is/")
    end)
    
    if success then
        local data = game:GetService("HttpService"):JSONDecode(result)
        return data
    end
    return nil
end
